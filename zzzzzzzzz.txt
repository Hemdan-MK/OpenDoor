        document.addEventListener('DOMContentLoaded', function () {
            // Initialize location buttons
            const locationButtonsContainer = document.getElementById('location-buttons');
            const locations = Object.keys(collegeData);

            // Create location buttons dynamically
            locations.forEach(location => {
                const button = document.createElement('button');
                button.classList.add('location-btn');
                button.setAttribute('data-location', location);
                button.textContent = location.charAt(0).toUpperCase() + location.slice(1);
                locationButtonsContainer.appendChild(button);
            });

            // Add event listeners to buttons
            const locationButtons = document.querySelectorAll('.location-btn');
            locationButtons.forEach(button => {
                button.addEventListener('click', function () {
                    // Remove active class from all buttons
                    locationButtons.forEach(btn => btn.classList.remove('active'));

                    // Add active class to clicked button
                    this.classList.add('active');

                    // Show colleges for selected location
                    displayColleges(this.getAttribute('data-location'));
                });
            });
        });

        function displayColleges(location) {
            const collegesDisplay = document.getElementById('colleges-display');
            const noSelectionMessage = document.getElementById('no-selection');
            const locationHeading = document.getElementById('location-heading');
            const collegeListContainer = document.getElementById('college-list');

            // Hide no selection message and show colleges display
            noSelectionMessage.style.display = 'none';
            collegesDisplay.style.display = 'block';

            // Update heading
            locationHeading.textContent = Top Colleges in ${location.charAt(0).toUpperCase() + location.slice(1)};

            // Clear previous college cards
            collegeListContainer.innerHTML = '';

            // Get colleges for the selected location
            const colleges = collegeData[location];

            // Create and append college cards
            colleges.forEach(college => {
                const collegeCard = document.createElement('div');
                collegeCard.classList.add('college-card');

                // Create logo container
                const logoContainer = document.createElement('div');
                logoContainer.classList.add('college-logo');

                // Create logo image
                const logoImg = document.createElement('img');
                logoImg.src = college.logo;
                logoImg.alt = ${college.name} Logo;
                if (college.logoScale) {
                    logoImg.style.transform = scale(${college.logoScale});
                }
                logoContainer.appendChild(logoImg);

                // Create info container
                const infoContainer = document.createElement('div');
                infoContainer.classList.add('college-info');

                // Create name
                const nameDiv = document.createElement('div');
                nameDiv.classList.add('college-name');
                nameDiv.textContent = college.name;

                // Create details
                const detailsDiv = document.createElement('div');
                detailsDiv.classList.add('college-details');
                detailsDiv.textContent = college.fullName;

                // Create description
                const descriptionP = document.createElement('p');
                descriptionP.classList.add('college-description');
                descriptionP.textContent = college.description;

                // Append elements to info container
                infoContainer.appendChild(nameDiv);
                infoContainer.appendChild(detailsDiv);
                infoContainer.appendChild(descriptionP);

                // Append logo and info to college card
                collegeCard.appendChild(logoContainer);
                collegeCard.appendChild(infoContainer);

                // Append college card to college list
                collegeListContainer.appendChild(collegeCard);
            });
        }

        document.addEventListener('DOMContentLoaded', function () {
            // Initialize location buttons
            const locationButtonsContainer = document.getElementById('location-buttons');
            const locations = Object.keys(collegeData);

            // Create location buttons dynamically
            locations.forEach(location => {
                const button = document.createElement('button');
                button.classList.add('location-btn');
                button.setAttribute('data-location', location);
                button.textContent = location.charAt(0).toUpperCase() + location.slice(1);
                locationButtonsContainer.appendChild(button);
            });

            // Add event listeners to buttons
            const locationButtons = document.querySelectorAll('.location-btn');
            locationButtons.forEach(button => {
                button.addEventListener('click', function () {
                    // Remove active class from all buttons
                    locationButtons.forEach(btn => btn.classList.remove('active'));

                    // Add active class to clicked button
                    this.classList.add('active');

                    // Show colleges for selected location
                    displayColleges(this.getAttribute('data-location'));
                });
            });
        });

        function displayColleges(location) {
            const collegesDisplay = document.getElementById('colleges-display');
            const noSelectionMessage = document.getElementById('no-selection');
            const locationHeading = document.getElementById('location-heading');
            const collegeListContainer = document.getElementById('college-list');

            // Hide no selection message and show colleges display
            noSelectionMessage.style.display = 'none';
            collegesDisplay.style.display = 'block';

            // Update heading
            locationHeading.textContent = Top Colleges in ${location.charAt(0).toUpperCase() + location.slice(1)};

            // Clear previous college cards
            collegeListContainer.innerHTML = '';

            // Get colleges for the selected location
            const colleges = collegeData[location];

            // Create and append college cards
            colleges.forEach(college => {
                const collegeCard = document.createElement('div');
                collegeCard.classList.add('college-card');

                // Create logo container
                const logoContainer = document.createElement('div');
                logoContainer.classList.add('college-logo');

                // Create logo image
                const logoImg = document.createElement('img');
                logoImg.src = college.logo;
                logoImg.alt = ${college.name} Logo;
                if (college.logoScale) {
                    logoImg.style.transform = scale(${college.logoScale});
                }
                logoContainer.appendChild(logoImg);

                // Create info container
                const infoContainer = document.createElement('div');
                infoContainer.classList.add('college-info');

                // Create name
                const nameDiv = document.createElement('div');
                nameDiv.classList.add('college-name');
                nameDiv.textContent = college.name;

                // Create details
                const detailsDiv = document.createElement('div');
                detailsDiv.classList.add('college-details');
                detailsDiv.textContent = college.fullName;

                // Create description
                const descriptionP = document.createElement('p');
                descriptionP.classList.add('college-description');
                descriptionP.textContent = college.description;

                // Append elements to info container
                infoContainer.appendChild(nameDiv);
                infoContainer.appendChild(detailsDiv);
                infoContainer.appendChild(descriptionP);

                // Append logo and info to college card
                collegeCard.appendChild(logoContainer);
                collegeCard.appendChild(infoContainer);

                // Append college card to college list
                collegeListContainer.appendChild(collegeCard);
            });
        }
    </script>
